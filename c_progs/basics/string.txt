#include<stdio.h>
void main()
{
    char a[20];
    scanf("%s",a);
    int count=0;
    int i;
    for(i=0;a[i]!='\0';i++)
    {
       count++;
    }
    for(i=0;a[i]!='\0';i++)
    {
        if(a[i]>='a' && a[i]<='z')
        a[i]-=32;
        printf("%c",a[i]);
    }
}

UPPER TO LOWER

#include<stdio.h>
void main()
{
    char a[20];
    scanf("%s",a);
    int count=0;
    int i;
    for(i=0;a[i]!='\0';i++)
    {
       count++;
    }
    for(i=count;i>=0;i--)
    {
        printf("%c",a[i]);
    }
}

REVERSE

#include<stdio.h>
void main()
{
    char a[20];
    scanf("%[^\n]%*c",a);
    int count=0;
    int i;
    for(i=0;a[i]!='\0';i++)
    {
       count++;
    }
    printf("%d",count);
}

    int i;
    for(i=0;a[i]!='\0';i++)
    {
        if(a[i] == ' ')
        {
            count++;
